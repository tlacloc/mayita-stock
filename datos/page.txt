    function countTime() {
        //获取当前时间

        var date = new Date();
        var now = getLocalTime(0,date);
        //var endDateStr="2020/3/26 06:00:00";
        var endDate = new Date(endDateStr).getTime();
        var leftTime = endDate-now;
        //时间差

        //定义变量 d,h,m,s保存倒计时的时间
        var d,h,m,s;
        if (leftTime>=0) {
            d = Math.floor(leftTime/1000/60/60/24);
            h = Math.floor(leftTime/(1000*60*60)%24);
            m = Math.floor(leftTime/1000/60%60);
            s = Math.floor(leftTime/1000%60);
            //将倒计时赋值到div中
            d = checkTime(d);
            h = checkTime(h);
            m = checkTime(m);
            s = checkTime(s);
            document.getElementById("_d").innerHTML = d;
            document.getElementById("_h").innerHTML = h;
            document.getElementById("_m").innerHTML = m;
            document.getElementById("_s").innerHTML = s;
            //递归每秒调用countTime方法，显示动态时间效果
            setTimeout(countTime,1000);
        }

    function checkTime(i) {
        if (i<10) {
            i = "0" +i;
        }
        return i;
    };


    function getLocalTime(i, d) {
        //参数i为时区值数字，比如北京为东八区则输入+8,西5输入-5
        //参数d为要转换的时间(Date类型)
        if (typeof i !== 'number') return;
        var len = d.getTime();
        //本地时间与GMT时间的时间偏移差
        var offset = d.getTimezoneOffset() * 60000;
        //得到现在的格林尼治时间
        var utcTime = len + offset;
        return utcTime + 3600000 * i;
    }
